Örnek Kod

import tensorflow as tf
from tensorflow.keras.applications import MobileNetV2
from tensorflow.keras.layers import Dense, GlobalAveragePooling2D
from tensorflow.keras.models import Model

# Transfer öğrenme için temel model
base_model = MobileNetV2(weights='imagenet', include_top=False, input_shape=(224, 224, 3))

# Temel model katmanlarını donduralım
for layer in base_model.layers:
    layer.trainable = False

# Model oluşturma
x = base_model.output
x = GlobalAveragePooling2D()(x)
x = Dense(1024, activation='relu')(x)
predictions = Dense(10, activation='softmax')(x)

model = Model(inputs=base_model.input, outputs=predictions)

# Model derleme
model.compile(
    optimizer='adam',
    loss='categorical_crossentropy',
    metrics=['accuracy']
)

Örnek Kod

import torch
from transformers import BertTokenizer, BertForSequenceClassification
from torch.utils.data import DataLoader, TensorDataset
from sklearn.metrics import accuracy_score, f1_score

# Tokenizer
tokenizer = BertTokenizer.from_pretrained('bert-base-uncased')

# Model
model = BertForSequenceClassification.from_pretrained('bert-base-uncased', num_labels=3)

# Metinleri tokenize etme ve model girdilerine dönüştürme
def preprocess_data(texts, labels=None):
    inputs = tokenizer(texts, padding=True, truncation=True, return_tensors="pt", max_length=128)
    if labels is not None:
        return TensorDataset(inputs['input_ids'], inputs['attention_mask'], torch.tensor(labels))
    return inputs

# Eğitim döngüsü (örnek)
def train(model, train_dataloader, optimizer, epochs=3):
    model.train()
    for epoch in range(epochs):
        for batch in train_dataloader:
            input_ids, attention_mask, labels = batch
            optimizer.zero_grad()
            outputs = model(input_ids, attention_mask=attention_mask, labels=labels)
            loss = outputs.loss
            loss.backward()
            optimizer.step()
            
# Duygu analizi fonksiyonu
def predict_sentiment(text, model, tokenizer):
    model.eval()
    inputs = tokenizer(text, return_tensors="pt", padding=True, truncation=True, max_length=128)
    with torch.no_grad():
        outputs = model(**inputs)
    probs = torch.nn.functional.softmax(outputs.logits, dim=1)
    prediction = torch.argmax(probs, dim=1).item()
    return ["Olumsuz", "Nötr", "Olumlu"][prediction]

Film Öneri Sistemi
Proje Detayları

Bu projede, kullanıcılara kişiselleştirilmiş film önerileri sunan hibrit bir öneri sistemi geliştirdim. İşbirlikçi filtreleme (collaborative filtering) ve içerik tabanlı filtreleme (content-based filtering) yöntemlerini birleştirerek hem kullanıcı tercihlerini hem de film özelliklerini değerlendiren bir sistem kurdum.

Python Scikit-Learn Pandas Flask
Durum: Tamamlandı
Tarih: Ocak 2025
Sistem Mimarisi

Öneri sistemim üç ana bileşenden oluşuyor:

  1. İşbirlikçi Filtreleme Modülü: Kullanıcı-film etkileşim matrisini kullanarak benzer beğenilere sahip kullanıcıları belirler ve matris faktorizasyonu (SVD) kullanarak önerilerde bulunur.
  2. İçerik Tabanlı Filtreleme Modülü: Film özelliklerini (tür, oyuncular, yönetmen, anahtar kelimeler) analiz ederek bir film vektör uzayı oluşturur ve kosinüs benzerliği kullanarak benzer filmleri önerir.
  3. Hibrit Entegrasyon Katmanı: İki modülden gelen sonuçları ağırlıklandırarak birleştirir ve nihai öneri listesini oluşturur.
Veri Seti

Sistem, MovieLens 25M veri seti üzerinde eğitildi. Bu veri seti, 160,000'den fazla kullanıcının 60,000'den fazla filme verdiği 25 milyon derecelendirmeyi içeriyor. Ek olarak, The Movie Database (TMDB) API'sinden film detaylarını çekerek içerik tabanlı filtreleme için zengin meta verileri elde ettim.

Örnek Kod

import pandas as pd
import numpy as np
from scipy.sparse.linalg import svds
from sklearn.metrics.pairwise import cosine_similarity
from sklearn.feature_extraction.text import TfidfVectorizer

# İşbirlikçi Filtreleme: SVD Modeli
def collaborative_filtering(ratings_matrix, user_id, num_recommendations=10):
    # SVD matris faktörizasyonu
    U, sigma, Vt = svds(ratings_matrix, k=50)
    sigma_diag = np.diag(sigma)
    
    # Tahmin matrisi oluşturma
    predicted_ratings = np.dot(np.dot(U, sigma_diag), Vt)
    
    # Kullanıcının henüz izlemediği filmleri öner
    user_row = ratings_matrix[user_id]
    unwatched_movies = np.where(user_row == 0)[0]
    
    # En yüksek tahmin puanlarına göre sırala
    user_predictions = predicted_ratings[user_id, unwatched_movies]
    recommended_indices = unwatched_movies[np.argsort(-user_predictions)][:num_recommendations]
    
    return recommended_indices

# İçerik Tabanlı Filtreleme
def content_based_filtering(movie_features, movie_id, num_recommendations=10):
    # TF-IDF vektörleştirme
    tfidf = TfidfVectorizer(stop_words='english')
    tfidf_matrix = tfidf.fit_transform(movie_features)
    
    # Kosinüs benzerliği hesaplama
    cosine_sim = cosine_similarity(tfidf_matrix[movie_id], tfidf_matrix)
    
    # En benzer filmleri sırala
    sim_scores = list(enumerate(cosine_sim[0]))
    sim_scores = sorted(sim_scores, key=lambda x: x[1], reverse=True)
    sim_scores = sim_scores[1:num_recommendations+1]  # İlk sırada filmin kendisi var
    movie_indices = [i[0] for i in sim_scores]
    
    return movie_indices

# Hibrit Öneri Sistemi
def hybrid_recommendations(user_id, movie_id, ratings_matrix, movie_features, collab_weight=0.7):
    # Her iki modelden önerileri al
    collab_recs = collaborative_filtering(ratings_matrix, user_id)
    content_recs = content_based_filtering(movie_features, movie_id)
    
    # Ağırlıklı birleştirme için skor hesaplama
    movie_scores = {}
    for i, movie_idx in enumerate(collab_recs):
        movie_scores[movie_idx] = collab_weight * (1.0 - i/len(collab_recs))
        
    for i, movie_idx in enumerate(content_recs):
        if movie_idx in movie_scores:
            movie_scores[movie_idx] += (1-collab_weight) * (1.0 - i/len(content_recs))
        else:
            movie_scores[movie_idx] = (1-collab_weight) * (1.0 - i/len(content_recs))
    
    # Skorlarına göre filmleri sırala
    ranked_movies = sorted(movie_scores.items(), key=lambda x: x[1], reverse=True)
    recommended_indices = [x[0] for x in ranked_movies]
    
    return recommended_indices

# Rasa Özel Eylem Örneği
from typing import Any, Text, Dict, List
from rasa_sdk import Action, Tracker
from rasa_sdk.executor import CollectingDispatcher
from rasa_sdk.events import SlotSet
import database_connector

class ActionOrderStatus(Action):
    def name(self) -> Text:
        return "action_order_status"
        
    def run(self, dispatcher: CollectingDispatcher,
            tracker: Tracker,
            domain: Dict[Text, Any]) -> List[Dict[Text, Any]]:
            
        order_number = tracker.get_slot("order_number")
        if not order_number:
            dispatcher.utter_message(template="utter_ask_order_number")
            return []
            
        # Veritabanından sipariş bilgilerini al
        order_info = database_connector.get_order_status(order_number)
        
        if not order_info:
            dispatcher.utter_message(text=f"Üzgünüm, {order_number} numaralı sipariş bulunamadı.")
            return []
            
        dispatcher.utter_message(
            text=f"{order_number} numaralı siparişiniz {order_info['status']} durumunda ve "
                 f"{order_info['estimated_delivery']} tarihinde teslim edilecek."
        )
        
        return [SlotSet("order_status", order_info['status'])]

Öngörülen Kod Yapısı

import torch
import cv2
import numpy as np
from time import time

# YOLOv5 modelini yükleme
model = torch.hub.load('ultralytics/yolov5', 'yolov5s')  # 'yolov5s', 'yolov5m', 'yolov5l', 'yolov5x'

# Tespit edilecek sınıfları belirleme (varsayılan olarak COCO veri seti sınıfları)
# classes = ['insan', 'araba', 'bisiklet', ... ]  # Özel sınıflar için
model.classes = None  # Tüm sınıfları tespit et

# Kamera akışını başlatma
cap = cv2.VideoCapture(0)  # 0: varsayılan kamera

# Performans metrikleri
fps_start_time = 0
fps = 0

while True:
    ret, frame = cap.read()
    if not ret:
        break
        
    # FPS hesaplama
    fps_end_time = time()
    time_diff = fps_end_time - fps_start_time
    if time_diff > 1.0:
        fps = 1.0 / time_diff
        fps_start_time = fps_end_time
        
    # Nesne tespiti
    results = model(frame)
    
    # Tespit sonuçlarını işleme
    detections = results.pandas().xyxy[0]  # Tespit sonuçlarını pandas DataFrame'ine dönüştürme
    
    # Tespit edilen nesneleri çerçeveleme ve etiketleme
    for i, detection in detections.iterrows():
        x1, y1, x2, y2 = int(detection['xmin']), int(detection['ymin']), int(detection['xmax']), int(detection['ymax'])
        conf = detection['confidence']
        cls = detection['class']
        label = f"{detection['name']} {conf:.2f}"
        
        # Sınırlayıcı kutu ve etiket çizme
        cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)
        cv2.putText(frame, label, (x1, y1 - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
    
    # FPS bilgisini ekrana yazma
    cv2.putText(frame, f"FPS: {fps:.2f}", (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 0, 255), 2)
    
    # Sonucu gösterme
    cv2.imshow('YOLOv5 Nesne Tespiti', frame)
    
    # Çıkış için 'q' tuşu kontrolü
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break
